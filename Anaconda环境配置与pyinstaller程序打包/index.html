    3. е…ұдә«еҢ…еӯҳеӮЁеә“пјҡиҷҪз„¶жҜҸдёӘ Conda зҺҜеўғжҳҜзӢ¬з«Ӣзҡ„пјҢдҪҶе®ғд»¬еҸҜд»Ҙе…ұдә«еҗҢдёҖдёӘ Conda еӯҳеӮЁеә“дёӯзҡ„еҢ…гҖӮд№ҹе°ұжҳҜиҜҙпјҢеҰӮжһңеӨҡдёӘзҺҜеўғйғҪйңҖиҰҒжҹҗдёӘзү№е®ҡзүҲжң¬зҡ„еҢ…пјҢConda еҸӘйңҖдёӢиҪҪдёҖж¬ЎиҜҘеҢ…е№¶е°Ҷе…¶еӯҳеӮЁеңЁе…¶еҢ…зј“еӯҳдёӯпјҢз„¶еҗҺеңЁдёҚеҗҢзҡ„зҺҜеўғдёӯй“ҫжҺҘеҲ°иҝҷдёӘеҢ…гҖӮиҝҷж ·еҸҜд»ҘиҠӮзңҒзЈҒзӣҳз©әй—ҙе’ҢдёӢиҪҪж—¶й—ҙгҖӮ

    4. и·ЁзҺҜеўғз§»еҠЁеҢ…пјҡдёҘж јжқҘиҜҙпјҢдҪ дёҚиғҪзӣҙжҺҘвҖңз§»еҠЁвҖқеҢ…д»ҺдёҖдёӘ Conda зҺҜеўғеҲ°еҸҰдёҖдёӘгҖӮдҪҶжҳҜпјҢйҖҡиҝҮдёҠиҝ°зҡ„зҺҜеўғеҜјеҮә / еҜје…ҘеҠҹиғҪпјҢдҪ еҸҜд»Ҙй«ҳж•Ҳең°еңЁдёҚеҗҢзҺҜеўғдёӯеӨҚеҲ¶еҢ…й…ҚзҪ®гҖӮ

    дәҢгҖҒPyinstaller дҪҝз”Ё

    дҪңз”ЁпјҡPyInstaller жҳҜдёҖдёӘйқһеёёжөҒиЎҢзҡ„ејҖжәҗе·Ҙе…·пјҢз”ЁдәҺе°Ҷ Python еә”з”ЁзЁӢеәҸжү“еҢ…жҲҗзӢ¬з«Ӣзҡ„еҸҜжү§иЎҢж–Үд»¶пјҢйҖӮз”ЁдәҺ WindowsгҖҒmacOS е’Ң Linux зӯүеӨҡдёӘе№іеҸ°гҖӮе®ғиғҪеӨҹеё®еҠ©ејҖеҸ‘иҖ…е°ҶеӨҚжқӮзҡ„йЎ№зӣ®еҸҠе…¶жүҖжңүдҫқиө–йЎ№ж•ҙеҗҲеҲ°дёҖиө·пјҢд»ҺиҖҢз®ҖеҢ–еҲҶеҸ‘жөҒзЁӢгҖӮ

    дё»иҰҒзү№зӮ№

    • и·Ёе№іеҸ°ж”ҜжҢҒпјҡеҸҜд»ҘеңЁдёҚеҗҢж“ҚдҪңзі»з»ҹдёҠиҝҗиЎҢпјҢз”ҹжҲҗеҜ№еә”е№іеҸ°зҡ„еҸҜжү§иЎҢж–Үд»¶гҖӮ
    • еҢ…еҗ«жүҖжңүдҫқиө–пјҡиҮӘеҠЁеҲҶжһҗдҪ зҡ„ Python и„ҡжң¬пјҢе№¶ж”¶йӣҶжүҖжңүзҡ„дҫқиө–еә“е’Ңиө„жәҗж–Үд»¶пјҢзЎ®дҝқеңЁзӣ®ж ҮжңәеҷЁдёҠж— йңҖйўқеӨ–е®үиЈ…еҚіеҸҜиҝҗиЎҢгҖӮ
    • еӨҡз§Қиҫ“еҮәж јејҸпјҡж №жҚ®дёҚеҗҢзҡ„йңҖжұӮпјҢеҸҜд»Ҙз”ҹжҲҗеҚ•дёӘеҸҜжү§иЎҢж–Үд»¶жҲ–еҢ…еҗ«еӨҡдёӘж–Үд»¶зҡ„зӣ®еҪ•гҖӮ
    • е…је®№жҖ§е№ҝпјҡж”ҜжҢҒеӨҡз§Қ Python зүҲжң¬еҸҠеӨ§йҮҸзҡ„з¬¬дёүж–№еә“гҖӮ

    2.1 е®үиЈ…дёҺд»Ӣз»Қ

    е®үиЈ… pyinstallerпјҢжЈҖжҹҘжҳҜеҗҰжңү pyinstallerпјҢеҰӮж— пјҢе®үиЈ… pip install pyinstaller

    Pyinstaller еёёз”ЁйҖүйЎ№

    • -F жҲ– --onefileпјҡжү“еҢ…жҲҗеҚ•дёӘеҸҜжү§иЎҢж–Үд»¶гҖӮ
    • -w жҲ– --windowedпјҲд»…йҖӮз”ЁдәҺ macOS е’Ң Windows дёҠзҡ„ GUI еә”з”ЁпјүпјҡйҒҝе…Қжү“ејҖжҺ§еҲ¶еҸ°зӘ—еҸЈгҖӮ
    • -n жҲ– --nameпјҡжҢҮе®ҡиҫ“еҮәзҡ„еҸҜжү§иЎҢж–Үд»¶еҗҚз§°гҖӮ
    • -i жҲ– --iconпјҡдёәз”ҹжҲҗзҡ„еҸҜжү§иЎҢж–Үд»¶ж·»еҠ еӣҫж ҮгҖӮ

    PyInstaller иҝҳе…Ғи®ёйҖҡиҝҮзј–еҶҷ .spec ж–Үд»¶жқҘиҝӣдёҖжӯҘе®ҡеҲ¶жү“еҢ…иҝҮзЁӢпјҢжҜ”еҰӮеӨ„зҗҶеҠЁжҖҒеҜје…ҘжЁЎеқ—гҖҒж·»еҠ ж•°жҚ®ж–Үд»¶зӯүеӨҚжқӮеңәжҷҜгҖӮ

    1. з”ҹжҲҗ .spec ж–Үд»¶пјҡ еңЁйҰ–ж¬ЎиҝҗиЎҢ PyInstaller ж—¶дёҚзӣҙжҺҘз”ҹжҲҗеҸҜжү§иЎҢж–Үд»¶пјҢиҖҢжҳҜеҸӘз”ҹжҲҗ .spec ж–Үд»¶пјҢеҸҜд»ҘдҪҝз”Ё -y еҸӮж•°пјҡ

      pyinstaller --onefile -n myapp -y myscript.py

      иҝҷе°Ҷз”ҹжҲҗдёҖдёӘеҗҚдёә myscript.spec зҡ„ж–Үд»¶гҖӮ

    2. зј–иҫ‘ .spec ж–Үд»¶пјҡ жү“ејҖз”ҹжҲҗзҡ„ .spec ж–Үд»¶иҝӣиЎҢзј–иҫ‘гҖӮд»ҘдёӢжҳҜдёҖдәӣеёёи§Ғзҡ„й«ҳзә§з”Ёжі•зӨәдҫӢгҖӮ

      # -*- mode: python ; coding: utf-8 -*-

#жҢҮе®ҡжҳҜеҗҰдҪҝз”ЁеҠ еҜҶз®—жі•
block_cipher = None

#жү“еҢ…жөҒзЁӢзҡ„ж ёеҝғйғЁеҲҶд№ӢдёҖгҖӮе®ғиҙҹиҙЈеҲҶжһҗдҪ зҡ„и„ҡжң¬еҸҠе…¶дҫқиө–е…ізі»пјҢе№¶еҮҶеӨҮжүҖжңүеҝ…иҰҒзҡ„дҝЎжҒҜд»ҘдҫҝеҗҺз»ӯжӯҘйӘӨдҪҝз”ЁгҖӮ
a = Analysis(['myscript.py'],  #иҰҒжү“еҢ…зҡ„дё»и„ҡжң¬еҲ—иЎЁ
    pathex=['/path/to/my/project'],  #еҢ…еҗ«йўқеӨ–зҡ„и·Ҝеҫ„д»ҘжҗңзҙўжЁЎеқ—
    binaries=[],  #йңҖиҰҒеҢ…еҗ«зҡ„дәҢиҝӣеҲ¶ж–Үд»¶пјҲеҰӮеҠЁжҖҒй“ҫжҺҘеә“пјү
    datas=[('resources/*', '.')], # ж·»еҠ ж•°жҚ®ж–Үд»¶
    hiddenimports=['module_name'], # жҳҫејҸеЈ°жҳҺйҡҗи—ҸеҜје…ҘпјҢйӮЈдәӣдёҚдјҡиў«иҮӘеҠЁжЈҖжөӢеҲ°зҡ„йҡҗејҸеҜје…ҘжЁЎеқ—
    hookspath=['./hooks'], # иҮӘе®ҡд№үй’©еӯҗи·Ҝеҫ„
    runtime_hooks=None,
    excludes=None,
    win_no_prefer_redirects=False,
    win_private_assemblies=False,
    cipher=block_cipher,
)

#Python ZIP еә”з”ЁзЁӢеәҸжЎЈжЎҲпјҢеҢ…еҗ«дәҶжүҖжңүз»ҸиҝҮеҺӢзј©зҡ„зәҜ Python жЁЎеқ—
pyz = PYZ(a.pure, a.zipped_data,
          cipher=block_cipher,
          )

#е®ҡд№үдәҶжңҖз»Ҳз”ҹжҲҗзҡ„еҸҜжү§иЎҢж–Үд»¶зҡ„иҜҰз»ҶдҝЎжҒҜгҖӮеҢ…жӢ¬еҗҚз§°гҖҒеӣҫж ҮгҖҒжҳҜеҗҰеҗҜз”ЁжҺ§еҲ¶еҸ°зӘ—еҸЈзӯүгҖӮ
exe = EXE(pyz,
          a.scripts,
          [],
          exclude_binaries=True,
          name='myapp',
          debug=False,
          strip=False,
          upx=True,
          console=True,
          icon='myicon.ico', # еә”з”Ёеӣҫж Ү
          key='my_secret_key') # еҠ еҜҶеҜҶй’Ҙ

#з”ЁдәҺж”¶йӣҶжүҖжңүйңҖиҰҒзҡ„иө„жәҗпјҢеҢ…жӢ¬дәҢиҝӣеҲ¶ж–Үд»¶гҖҒZIP ж–Үд»¶е’Ңж•°жҚ®ж–Үд»¶пјҢе№¶е°Ҷе®ғд»¬дёҺ exe з»„еҗҲеңЁдёҖиө·еҪўжҲҗжңҖз»Ҳзҡ„иҫ“еҮәзӣ®еҪ•з»“жһ„гҖӮ
coll = COLLECT(exe,
               a.binaries,
               a.zipfiles,
               a.datas,
               strip=False,
               upx=True,
               upx_exclude=[],
               name='myapp')

    2.3 е…¶д»–зӣёе…ізҹҘиҜҶзӮ№

    python -m venv е’Ң AnacondaпјҲжҲ–жӣҙе…·дҪ“ең°иҜҙжҳҜ conda е‘Ҫд»Өз”ЁдәҺеҲӣе»әиҷҡжӢҹзҺҜеўғпјүйғҪжҳҜз”ЁдәҺеҲӣе»әзӢ¬з«Ӣзҡ„ Python зҺҜеўғпјҢдҪҶе®ғд»¬д№Ӣй—ҙеӯҳеңЁдёҖдәӣе…ій”®е·®ејӮпјҡ

    1. еҢ…з®ЎзҗҶ
    • venv: дҪҝз”Ёж ҮеҮҶзҡ„ Python еҢ…з®ЎзҗҶе·Ҙе…· pip жқҘе®үиЈ…е’Ңз®ЎзҗҶеҢ…гҖӮиҝҷж„Ҹе‘ізқҖдҪ йңҖиҰҒжүӢеҠЁи§ЈеҶідҫқиө–е…ізі»пјҢе№¶зЎ®дҝқжүҖжңүйңҖиҰҒзҡ„еҢ…йғҪеҸҜд»ҘйҖҡиҝҮ PyPI иҺ·еҫ—гҖӮ
    • Anaconda (conda): жҸҗдҫӣдәҶиҮӘе·ұзҡ„еҢ…з®ЎзҗҶе’ҢзҺҜеўғз®ЎзҗҶзі»з»ҹпјҢеҸ«еҒҡ condaгҖӮе®ғдёҚд»…иғҪеӨҹз®ЎзҗҶ Python еҢ…пјҢиҝҳиғҪеӨ„зҗҶйқһ Python дҫқиө–йЎ№пјҢдҫӢеҰӮзј–иҜ‘еҷЁгҖҒзі»з»ҹеә“зӯүгҖӮжӯӨеӨ–пјҢconda еҸҜд»Ҙд»Һ Anaconda зҡ„д»“еә“дёӯиҺ·еҸ–еҢ…пјҢиҝҷдәӣд»“еә“йҖҡеёёеҢ…еҗ«жҜ” PyPI жӣҙж–°зҡ„зүҲжң¬жҲ–иҖ…зӣҙжҺҘжҸҗдҫӣйў„зј–иҜ‘зҡ„дәҢиҝӣеҲ¶ж–Үд»¶пјҢиҝҷеңЁжҹҗдәӣжғ…еҶөдёӢеҸҜд»Ҙз®ҖеҢ–е®үиЈ…иҝҮзЁӢгҖӮ
    1. и·Ёе№іеҸ°е…је®№жҖ§дёҺжҳ“з”ЁжҖ§
    • venv: дё»иҰҒдё“жіЁдәҺ Python ејҖеҸ‘иҖ…зҡ„дҪҝз”ЁеңәжҷҜпјҢеҜ№дәҺз®ҖеҚ•зҡ„йЎ№зӣ®е’ҢзәҜ Python дҫқиө–йқһеёёжңүж•ҲгҖӮдҪҶжҳҜпјҢеңЁеӨ„зҗҶи·Ёе№іеҸ°е…је®№жҖ§й—®йўҳж—¶еҸҜиғҪйңҖиҰҒйўқеӨ–зҡ„е·ҘдҪңпјҢзү№еҲ«жҳҜеңЁеӨ„зҗҶдёҚеҗҢж“ҚдҪңзі»з»ҹд№Ӣй—ҙзҡ„е·®ејӮж—¶гҖӮ
    • Anaconda (conda): и®ҫи®Ўж—¶иҖғиҷ‘еҲ°дәҶж•°жҚ®з§‘еӯҰйўҶеҹҹзҡ„йңҖжұӮпјҢеӣ жӯӨеңЁеӨ„зҗҶи·Ёе№іеҸ°е…је®№жҖ§е’ҢеӨҚжқӮзҡ„дҫқиө–е…ізі»ж–№йқўиЎЁзҺ°еҮәиүІгҖӮconda еҲӣе»әзҡ„зҺҜеўғжӣҙе®№жҳ“еңЁдёҚеҗҢзҡ„ж“ҚдҪңзі»з»ҹд№Ӣй—ҙиҝҒз§»пјҢдё”ж”ҜжҢҒеӨҡз§ҚиҜӯиЁҖзҡ„еҢ…з®ЎзҗҶпјҢдёҚд»…д»…йҷҗдәҺ PythonгҖӮ
    1. зҺҜеўғйҡ”зҰ»
    • venv е’Ң conda йғҪиғҪжңүж•Ҳең°еҲӣе»әйҡ”зҰ»зҡ„зҺҜеўғпјҢдҪҝеҫ—дёҚеҗҢзҡ„йЎ№зӣ®еҸҜд»ҘжӢҘжңүзӢ¬з«Ӣзҡ„еә“зүҲжң¬е’Ңдҫқиө–е…ізі»гҖӮз„¶иҖҢпјҢз”ұдәҺ conda иҝҳеҸҜд»Ҙз®ЎзҗҶйқһ Python дҫқиө–пјҢжүҖд»ҘеңЁжҹҗдәӣжғ…еҶөдёӢе®ғеҸҜд»ҘжҸҗдҫӣжӣҙејәзҡ„йҡ”зҰ»иғҪеҠӣгҖӮ
    1. жҖ§иғҪдёҺеӨ§е°Ҹ
    • venv: з”ҹжҲҗзҡ„иҷҡжӢҹзҺҜеўғзӣёеҜ№иҫғе°ҸпјҢеӣ дёәе®ғд»¬еҸӘеҢ…еҗ«еҝ…иҰҒзҡ„ Python ж–Үд»¶е’ҢжҝҖжҙ»и„ҡжң¬гҖӮ
    • Anaconda (conda): еҲӣе»әзҡ„зҺҜеўғеҸҜиғҪдјҡжӣҙеӨ§пјҢеӣ дёәе®ғеҢ…еҗ«дәҶжӣҙеӨҡзҡ„й»ҳи®ӨеҢ…пјҲе°Өе…¶жҳҜж•°жҚ®з§‘еӯҰзӣёе…ізҡ„еә“пјүпјҢд»ҘеҸҠдёәдәҶж”ҜжҢҒеӨҡиҜӯиЁҖе’ҢеӨҚжқӮдҫқиө–иҖҢеўһеҠ зҡ„еҠҹиғҪгҖӮ
    ж¬ўиҝҺеҗ„дҪҚзңӢе®ҳеҸҠжҠҖжңҜеӨ§дҪ¬еүҚжқҘдәӨжөҒжҢҮеҜје‘ҖпјҢеҸҜд»ҘйӮ®д»¶иҮі jqiange@yeah.net
